Task 1: Rectangle Area

Write a function area(length, width) that returns the area of a rectangle.

function area(length, width) {
  return length * width;
}

console.log(area(5, 3));   // 15
console.log(area(10, 4));  // 40
console.log(area(7, 7));   // 49

Bonus: Rewrite as an arrow function with implicit return:

let areaArrow = (length, width) => length * width;

console.log(areaArrow(8, 6));  // 48

Task 2: Greeting with Default

                    
Create a function that greets a user, with a default greeting if no name is provided.

                    
function sayHello(name = "Guest") {
  return "Hello, " + name + "!";
}

console.log(sayHello("Priya"));
console.log(sayHello());  // Uses default

Task 3: Multi-step Calculation

                    
function calculateBill(price, taxPercent) {
  let tax = (price * taxPercent) / 100;
  let total = price + tax;
  return total;
}

let bill = calculateBill(1000, 18);
console.log("Total with 18% GST:  + bill);
